    <script>
        // --- 1. Inicialização ---
        $(document).ready(function() {
            // Inicializa Select2
            $('.select2-padrao').select2({
                width: '100%'
            });
        });

        // --- 2. Controle de Navegação (SPA) ---
        const mapTelas = {
            'menu': 'menuInicial',
            'novo_usuario': 'telaCadastroUsuario',
            'auditoria': 'telaAuditoria', // Nova rota
            'ajuda': 'telaAjuda'
        };

        // Adiciona evento de clique em todos os links do menu
        document.querySelectorAll('.sidebar nav a[data-action]').forEach(link => {
            link.addEventListener('click', function(e) {
                e.preventDefault();
                const acao = this.getAttribute('data-action');

                if (acao === 'sair') {
                    if(confirm("Deseja realmente sair?")) {
                        window.location.reload();
                    }
                    return;
                }

                if (mapTelas[acao]) {
                    // Esconde todas as telas
                    document.querySelectorAll('.tela').forEach(t => t.style.display = 'none');
                    // Mostra a selecionada
                    document.getElementById(mapTelas[acao]).style.display = 'block';

                    // Atualiza classe active no menu
                    document.querySelectorAll('.sidebar nav a').forEach(l => l.classList.remove('active'));
                    this.classList.add('active'); // Adiciona ao clicado
                    
                    // Tratamento de submenus
                    const parentLi = this.closest('.menu-group');
                    if(parentLi) {
                        const parentLink = parentLi.querySelector('.menu-toggle');
                    }

                    // Se for auditoria, carrega os dados (que agora virá vazio inicialmente)
                    if(acao === 'auditoria') {
                        carregarAuditoria();
                    }
                }
            });
        });

        // --- 3. Sidebar Toggle ---
        const btnToggle = document.getElementById('btnToggleSidebar');
        const sidebar = document.querySelector('.sidebar');
        
        btnToggle.addEventListener('click', () => {
            sidebar.classList.toggle('collapsed');
        });

        // --- 4. Submenus (Accordion) ---
        document.querySelectorAll('.menu-toggle').forEach(item => {
            item.addEventListener('click', function(e) {
                e.preventDefault();
                this.parentElement.classList.toggle('open');
            });
        });

        // --- 5. Funções de Negócio ---

        // Função para Salvar Usuário (Conecta com a classe Usuario.java)
        function salvarUsuario() {
            // Captura os dados do formulário
            const nome = $('#userNome').val();
            const email = $('#userEmail').val();
            const senha = $('#userSenha').val();
            const telefone = $('#userTelefone').val();
            const status = $('#userStatus').val();

            // Validação simples
            if (!nome || !email || !senha) {
                mostrarModal("Erro", "Por favor, preencha os campos obrigatórios (Nome, Email e Senha).");
                return;
            }

            // Objeto Usuario (DTO) pronto para ser enviado como JSON
            const usuarioDTO = {
                nome: nome,
                email: email,
                senha: senha, 
                numeroTelefone1: telefone,
                userStatus: status
            };

            // AJAX: Envio para o Backend
            $.ajax({
                url: '/api/usuarios', // Endpoint fictício
                method: 'POST',
                contentType: 'application/json',
                data: JSON.stringify(usuarioDTO),
                success: function(response) {
                    mostrarModal("Sucesso", "Usuário cadastrado com sucesso!");
                    // Limpar formulário
                    $('#userNome').val('');
                    $('#userEmail').val('');
                    $('#userSenha').val('');
                    $('#userTelefone').val('');
                },
                error: function(xhr, status, error) {
                    console.log(error);
                    // Apenas para visualização (mock)
                    mostrarModal("Simulação", "JSON gerado e pronto para envio: " + JSON.stringify(usuarioDTO));
                }
            });
        }

        // Função para Carregar Auditoria (Sem exemplos, apenas estrutura)
        function carregarAuditoria() {
            // Captura filtros
            const filtroNome = $('#filtroAuditUser').val();
            const filtroAcao = $('#filtroAuditAcao').val();

            // Renderização da tabela (Limpeza)
            const tbody = document.getElementById('tbodyAuditoria');
            tbody.innerHTML = ''; // Garante que a tabela esteja vazia

            /* AQUI FUTURAMENTE ENTRARÁ O CÓDIGO AJAX PARA BUSCAR DADOS REAIS:
            $.get('/api/movimentacoes', { nome: filtroNome, acao: filtroAcao }, function(data) { ... });
            */
        }

        // Modal Customizado
        function mostrarModal(titulo, msg) {
            document.getElementById('modalTitulo').innerText = titulo;
            document.getElementById('modalMensagem').innerText = msg;
            document.getElementById('modalGlobal').style.display = 'block';
        }

        function fecharModal() {
            document.getElementById('modalGlobal').style.display = 'none';
        }
